    <div class="buttons">
      <button id="prevBtn" disabled>السابق</button>
      <button id="nextBtn">التالي</button>
    </div>
  </div>

  <script>
    const lessons = [
      "مرحباً بك في تطبيق التعليم الذاتي! هذا هو الدرس الأول.",
      "الدرس الثاني: التنظيم الذاتي يساعدك على التعلم بشكل أفضل.",
      "الدرس الثالث: استخدم أدوات التخطيط لتسهيل تنظيم وقتك.",
      "الدرس الرابع: الممارسة المنتظمة تحسن التحصيل الأكاديمي.",
      "الدرس الخامس: لا تتردد في طلب المساعدة عندما تحتاجها.",
      "الدرس السادس: استخدم تقنيات الاستذكار الفعالة لتعزيز حفظ المعلومات.",
      "الدرس السابع: قيّم تقدمك بانتظام لتعرف نقاط قوتك وضعفك.",
      "الدرس الثامن: حافظ على تحفيزك وتحدى نفسك دائماً."
    ];

    let currentIndex = -1;
    const lessonDiv = document.getElementById("lesson");
    const prevBtn = document.getElementById("prevBtn");
    const nextBtn = document.getElementById("nextBtn");
    const progressText = document.getElementById("progress");

    function updateProgress() {
      if(currentIndex === -1){
        progressText.textContent = "لم تبدأ الدروس بعد.";
      } else {
        progressText.textContent = `الدرس ${currentIndex + 1} من ${lessons.length}`;
      }
    }

    function showLesson() {
      if(currentIndex === -1){
        lessonDiv.textContent = "اضغط التالي لبدء الدرس.";
        prevBtn.disabled = true;
        nextBtn.disabled = false;
      } else {
        lessonDiv.textContent = lessons[currentIndex];
        prevBtn.disabled = currentIndex === 0;
        nextBtn.disabled = currentIndex === lessons.length - 1;
      }
      updateProgress();
    }

    prevBtn.addEventListener("click", () => {
      if(currentIndex > 0){
        currentIndex--;
        showLesson();
      }
    });

    nextBtn.addEventListener("click", () => {
      if(currentIndex < lessons.length -1){
        currentIndex++;
        showLesson();
      }
    });

    showLesson();
  </script>
